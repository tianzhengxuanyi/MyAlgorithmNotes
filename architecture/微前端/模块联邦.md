<!-- 处理暴露出去的页面 -->

```js

// 存储暴露给模块联邦的页面路径映射
const exposePages = {};

/**
 * 根据页面路径获取实际文件路径
 * @param {string} page - 页面路径(可能包含参数)
 * @returns {string|null} 返回实际文件路径或null(如果文件不存在)
 */
const getPagePath = function (page: string) {
  // 处理带参数的路径
  if (page.includes("?")) {
    page = page.split("?")[0];
  }
  const prefix = "./src/views/";
  
  // 处理不带扩展名的路径
  if (!page.endsWith(".vue") && !page.endsWith(".ts")) {
    const pageExt = ["vue", "ts", "tsx", "js"]; // 支持的页面扩展名
    
    // 尝试追加扩展名查找文件
    for (let i = 0; i < pageExt.length; i++) {
      const ext = pageExt[i];
      if (fs.existsSync(prefix + page + "." + ext)) {
        return prefix + page + "." + ext;
      }
    }
    
    // 尝试查找index文件
    for (let i = 0; i < pageExt.length; i++) {
      const ext = pageExt[i];
      if (fs.existsSync(prefix + page + "/index." + ext)) {
        return prefix + page + "/index." + ext;
      }
    }
    return null;
  }
  
  // 直接检查带扩展名的路径
  if (fs.existsSync(prefix + page)) {
    return prefix + page;
  }
  return null;
};

/**
 * 添加页面到模块联邦暴露列表
 * @param {Array} pages - 页面数组，包含id和component属性
 * @param {PluginOption} option - 插件配置选项
 */
const addPages = function (
  pages: { id: string; component: string }[],
  option: PluginOption
) {
  pages.forEach((item: { id: string; component: string }) => {
    let page = item.component;
    
    // 处理带参数的路径
    if (page.includes("?")) {
      page = page.split("?")[0];
    }
    const rawPage = page;
    
    // 检查是否为子系统且配置了模块名
    if (!option.isUnionSub || !option.module) {
      return;
    }
    
    // 检查页面是否属于当前模块
    if (!page.startsWith(option.module + "/")) {
      return;
    }
    
    // 再次处理带参数的路径(冗余处理)
    if (page.includes("?")) {
      page = page.split("?")[0];
    }
    
    // 获取实际文件路径
    const pagePath = getPagePath(page);
    
    // 检查文件是否存在
    if (!pagePath) {
      console.log("文件不存在:", page);
      return;
    }
    
    console.log(rawPage); 
    // 将页面添加到暴露列表，key为"./原始路径"
    exposePages["./" + rawPage] = pagePath;
  });
};

```


## 模块联邦升级问题

#### @antv/x6自定义节点不显示

@antv/x6的依赖需要共享,尤其是@antv/x6-vue-shape,需要在主系统和子系统都安装,并共享依赖

远程依赖中导入方式

```js
import {register} from '@antv/x6-vue-shape' // √

import {register} from '@antv/x6-vue-shape/src/index' // × 这会导致把@antv/x6-vue-shape/src/index打包到对应文件中，而不会共享依赖
```

#### el-tabs active不切换

elementui-plus需要共享，且@vueuse/core也需要。elementui-plus里面依赖了@vueuse/core,所以需要共享。
