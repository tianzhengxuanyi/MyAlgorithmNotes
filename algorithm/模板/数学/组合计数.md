### 组合计数

**组合数**
```js
const MOD = BigInt(1e9 + 7);
const MX = 100001;

const f = Array(MX);
f[0] = 1n;

for (let i = 1; i < MX; i++) {
    f[i] = f[i - 1] * BigInt(i) % MOD;
}

const pow = (x, n) => {
    x = BigInt(x);
    let res = 1n;
    while (n) {
        if (n & 1n) {
            res = res * x % MOD;
        }
        x = x * x % MOD;
        n = n >> 1n;
    }
    return res;
}
const invF = Array(MX);
invF[MX - 1] = pow(f[MX - 1], MOD - 2n);

for (let i = MX - 2; i >= 0; i--) {
    invF[i] = invF[i + 1] * BigInt(i + 1) % MOD;
}


const comb = (n, k) => {
    return f[n] * invF[k] * invF[n - k];
}

```

**杨辉三角**
```js
// 组合数学解法：C(k, m) 其中m = (必须步数 + 剩余步数/2)
const C = Array.from({ length: k + 1 }, () => Array(k + 1).fill(0));

// 构建组合数表（帕斯卡三角形）
for (let i = 0; i <= k; i++) {
    C[i][0] = 1; // 任何数选0个的方案都是1
    for (let j = 1; j <= i; j++) {
        // 组合数递推公式，结果取模
        C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]);
    }
}
```
